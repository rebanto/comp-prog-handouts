\begin{enumerate}[label=\textbf{\arabic*.}]
    \item You are given an array of size $n$. A loop runs once through the array and prints each element. What is the time complexity?  
    \vspace{2cm}

    \item A nested loop is written where the outer loop runs $n$ times and the inner loop also runs $n$ times. What is the time complexity?  
    \vspace{2cm}

    \item You sort an array of $n$ numbers using Merge Sort. State the time complexity.  
    \vspace{2cm}

    \item Consider the following pseudocode:
    \begin{lstlisting}
for i in range(1, n+1):
    j = i
    while j > 0:
        j = j // 2
    \end{lstlisting}
    Determine the time complexity.  
    \vspace{2.5cm}

    \item A number $n$ is tested for primality by dividing it by every integer from $2$ to $\sqrt{n}$. What is the time complexity?  
    \vspace{2cm}

    \item Suppose you perform $m$ operations, and each operation requires $O(\log n)$ time (such as in a balanced binary search tree). Express the total time complexity.  
    \vspace{2cm}

    \item Analyze the following pseudocode:
    \begin{lstlisting}
for i in range(1, n+1):
    for j in range(1, i+1):
        print(i, j)
    \end{lstlisting}
    What is the time complexity?  
    \vspace{2.5cm}

    \item Consider the loop:
    \begin{lstlisting}
i = 1
while i <= n:
    i = i * 2
    \end{lstlisting}
    What is the time complexity?  
    \vspace{2.5cm}

    \item A recursive function is defined as:
    \begin{lstlisting}
def f(n):
    if n == 1:
        return
    f(n-1)
    f(n-1)
    \end{lstlisting}
    What is the time complexity?  
    \vspace{2.5cm}

    \item Evaluate the following nested loops:
    \begin{lstlisting}
for i in range(1, n+1):
    for j in range(1, int(sqrt(n))+1):
        print(i, j)
    \end{lstlisting}
    What is the time complexity?  
    \vspace{2.5cm}
\end{enumerate}
