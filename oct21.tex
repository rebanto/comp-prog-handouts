\section*{Describe the Solution Set}
For each of the following problems, describe the solution set. \textit{\textbf{Do not write code to solve the problem.}}

\begin{enumerate}[label=\textbf{P\arabic*.}, itemsep=1.4in]
    \item Find all integer pairs $(x,y)$ with $0 \le x,y \le 10$ that satisfy $x^2 + y^2 = 25$.
    \item Find all 3-digit numbers whose digits are strictly increasing.
    \item Find all length-8 binary strings with no two 1's next to each other. 
    \item Find all subsets of $\{1,2,3,4\}$ whose sum is greater than 5. 
    \item Find all permutations of the letters \{A,B,C,D\} where A comes before B.
    \item Find all 4-digit numbers where the digits add up to 10. 
    \item Find all pairs $(a,b)$ from $\{1,2,\dots,10\}$ such that $a<b$ and $a+b$ is even.
    \item Find all subsets of $\{1,2,\dots,8\}$ that contain no two consecutive numbers.
    \item Find all 3-digit numbers that are palindromes.
    \item Find all integer triples $(x,y,z)$ with $1 \le x,y,z \le 5$ such that $x+y+z$ is even.
\end{enumerate}
